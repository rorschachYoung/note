## 2.Node中使用redis
```js
const redis = require('redis')

const redisClient = redis.createClient(6379,'127.0.0.1')
redisClient.on('err',err => {
    console.log(err)
})
redisClient.set('name','haha',redis.print /* 会把结果打印在控制台 */)
redisClient.get('name',(err,v)=>{
    if(err){
        console.log(err)
        redisClient.quit()
        return
    }
    console.log(v)
    /* 退出 */
    redisClient.quit()
})
```
